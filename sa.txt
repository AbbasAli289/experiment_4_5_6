//swapping of 2 number using 3rd variable.
#include<stdio.h>
int main()
{
int a=10,b=20,c;
printf("before swapping a=%d b=%d",a,b);
c=a;
a=b;
b=c;
printf("after swapping a=%d b=%d",a,b);
}
